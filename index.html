<script>

    const username = "abrar-sprinklr"; // your GitHub username

    const repo = "pif-reports.github.io"; // your repo name

    const folderPath = "reports"; // folder where reports are stored


    async function fetchReports() {

        const reportList = document.getElementById("report-list");

        reportList.innerHTML = "<li>Loading reports...</li>";


        try {

            const response = await fetch(`https://api.github.com/repos/${username}/${repo}/contents/${folderPath}`);

            if (!response.ok) throw new Error("GitHub API request failed");


            const files = await response.json();

            const docxFiles = files.filter(file => file.name.toLowerCase().endsWith(".docx"));


            if (docxFiles.length === 0) {

                reportList.innerHTML = "<li>No reports found.</li>";

                return;

            }


            // Sort files by name (latest first if filenames are chronological)

            docxFiles.sort((a, b) => b.name.localeCompare(a.name));


            // Clear list and populate

            reportList.innerHTML = "";

            docxFiles.forEach(file => {

                const li = document.createElement("li");

                const link = document.createElement("a");

                link.href = file.download_url;

                link.target = "_blank";

                link.textContent = file.name.replace(/_/g, " ").replace(".docx", "");

                li.appendChild(link);

                reportList.appendChild(li);

            });

        } catch (error) {

            console.error(error);

            reportList.innerHTML = "<li>Error loading reports. Please try again later.</li>";

        }

    }


    fetchReports();

</script>
